% sidebar_template.tex

\documentclass[12pt]{article}

\usepackage[letterpaper, margin=1in]{geometry} % Set page size and margins
\usepackage{paracol}  % Package for parallel columns
\usepackage[breaklinks, colorlinks, linkcolor=blue]{hyperref}
\usepackage{titlesec} % Package for modifying section titles
\usepackage[normalem]{ulem} % For handling underlines, [normalem] disables default underlining

\usepackage{fancyhdr}
\pagestyle{fancy}  % Set page style to 'fancy'
\fancyhf{}  % Clear existing header and footer definitions
\fancyfoot[R]{\thepage}  % Place page number on the right side of the footer
\renewcommand{\headrulewidth}{0pt}  % Remove header rule (optional)
\renewcommand{\footrulewidth}{0pt}  % Remove footer rule (optional)


% Define the 'narrow' environment
\newenvironment{narrow}{%
  \list{}{%
    \setlength{\leftmargin}{0em}% Adjust the left margin as needed
    \setlength{\rightmargin}{9em}% Adjust the right margin as needed
  }%
\item\relax
}{%
  \endlist
}

% Define the 'wide' environment
\newenvironment{wide}{%
  \list{}{%
    \setlength{\leftmargin}{-6em}% Adjust the left margin as needed
    \setlength{\rightmargin}{0em}% Adjust the right margin as needed
  }%
\item\relax
}{%
  \endlist
}

% Define the '\tightlist' command
\providecommand{\tightlist}{%
  \setlength{\itemsep}{0pt}\setlength{\parskip}{0pt}}

% Redefine sectioning commands to remove numbering
\titleformat{\section}[hang]{\normalfont\Large\bfseries}{}{0em}{}
\titleformat{\subsection}[hang]{\normalfont\large\bfseries}{}{0em}{}
\titleformat{\subsubsection}[hang]{\normalfont\normalsize\bfseries}{}{0em}{}
\titleformat{\paragraph}[hang]{\normalfont\normalsize\bfseries}{}{0em}{}
\titleformat{\subparagraph}[hang]{\normalfont\normalsize\bfseries}{}{0em}{}


% Adjust the space before and after titles
%\titlespacing*{\section}{0pt}{2em}{1em}
\titlespacing*{\section}{0pt}{1.5em}{0.75em}
\titlespacing*{\subsection}{0pt}{1.25em}{0.5em}

%\titlespacing*{\paragraph}{0pt}{1em}{0.5em}
%\titlespacing*{\subparagraph}{0pt}{0.75em}{0.5em}

% Adjust the space between paragraphs
\setlength{\parskip}{0.5em} % Adjust this length as needed for the desired space

\begin{document}

\begin{flushleft}
  \input{main.tex}
\end{flushleft}

\end{document}
